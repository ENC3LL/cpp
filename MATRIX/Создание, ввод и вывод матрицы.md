# Создание, ввод и вывод матрицы

Это основа для работы. Матрицу нужно уметь создавать по заданным размерам, заполнять с клавиатуры и красиво выводить на экран.

```c++
#include <iostream>
#include <vector>
#include <iomanip> // для std::setw

// Функция для удобного вывода матрицы
void printMatrix(const std::vector<std::vector<int>>& matrix) {
    if (matrix.empty()) {
        std::cout << "Матрица пуста!" << std::endl;
        return;
    }
    for (const auto& row : matrix) { // Перебор строк
        for (int element : row) {    // Перебор элементов в строке
            std::cout << std::setw(4) << element << " "; // setw для выравнивания
        }
        std::cout << std::endl;
    }
}

int main() {
    int rows, cols;
    std::cout << "Введите количество строк матрицы: ";
    std::cin >> rows;
    std::cout << "Введите количество столбцов матрицы: ";
    std::cin >> cols;

    // Создаем матрицу нужного размера
    std::vector<std::vector<int>> matrix(rows, std::vector<int>(cols));

    // Заполняем матрицу с клавиатуры
    std::cout << "Введите элементы матрицы (" << rows << "x" << cols << "):" << std::endl;
    for (int i = 0; i < rows; ++i) {
        std::cout << "Строка " << i << ": ";
        for (int j = 0; j < cols; ++j) {
            std::cin >> matrix[i][j];
        }
    }

    // Выводим матрицу
    std::cout << "\nВаша матрица:" << std::endl;
    printMatrix(matrix);

    return 0;
}
```
