# Сложение и вычитание матриц

Складывать можно только матрицы одинакового размера. Сложение происходит поэлементно.

```c++
#include <iostream>
#include <vector>

// ... (вставьте сюда функцию printMatrix из предыдущего примера) ...

// Функция сложения двух матриц
std::vector<std::vector<int>> addMatrices(const std::vector<std::vector<int>>& A, const std::vector<std::vector<int>>& B) {
    // Проверка на совпадение размеров
    if (A.size() != B.size() || A[0].size() != B[0].size()) {
        std::cout << "Ошибка: Матрицы имеют разные размеры!" << std::endl;
        return {}; // Возвращаем пустую матрицу в случае ошибки
    }

    int rows = A.size();
    int cols = A[0].size();
    std::vector<std::vector<int>> result(rows, std::vector<int>(cols));

    for (int i = 0; i < rows; ++i) {
        for (int j = 0; j < cols; ++j) {
            result[i][j] = A[i][j] + B[i][j]; // Складываем соответствующие элементы
        }
    }
    return result;
}

int main() {
    std::vector<std::vector<int>> matrixA = {{1, 2}, {3, 4}};
    std::vector<std::vector<int>> matrixB = {{5, 6}, {7, 8}};

    std::vector<std::vector<int>> result = addMatrices(matrixA, matrixB);

    std::cout << "Результат сложения:" << std::endl;
    printMatrix(result);

    return 0;
}
```
