# Умножение матриц

Классическая операция, где количество столбцов первой матрицы должно быть равно количеству строк второй.

```c++
#include <iostream>
#include <vector>
#include <iomanip>

// Функция для вывода матрицы
void printMatrix(const std::vector<std::vector<int>>& matrix) {
    if (matrix.empty()) return;
    for (const auto& row : matrix) {
        for (int element : row) std::cout << std::setw(5) << element << " ";
        std::cout << std::endl;
    }
}

int main() {
    int rowsA, colsA, rowsB, colsB;
    std::cout << "Введите размеры матрицы A (строки столбцы): ";
    std::cin >> rowsA >> colsA;
    std::cout << "Введите размеры матрицы B (строки столбцы): ";
    std::cin >> rowsB >> colsB;

    if (colsA != rowsB) {
        std::cout << "Умножение невозможно! Количество столбцов A должно быть равно количеству строк B." << std::endl;
        return 1;
    }

    std::vector<std::vector<int>> matrixA(rowsA, std::vector<int>(colsA));
    std::cout << "Введите элементы матрицы A:" << std::endl;
    for (int i = 0; i < rowsA; ++i) for (int j = 0; j < colsA; ++j) std::cin >> matrixA[i][j];

    std::vector<std::vector<int>> matrixB(rowsB, std::vector<int>(colsB));
    std::cout << "Введите элементы матрицы B:" << std::endl;
    for (int i = 0; i < rowsB; ++i) for (int j = 0; j < colsB; ++j) std::cin >> matrixB[i][j];

    // Результирующая матрица будет иметь размер rowsA x colsB
    std::vector<std::vector<int>> resultMatrix(rowsA, std::vector<int>(colsB, 0));

    for (int i = 0; i < rowsA; ++i) {
        for (int j = 0; j < colsB; ++j) {
            for (int k = 0; k < colsA; ++k) { // colsA или rowsB
                resultMatrix[i][j] += matrixA[i][k] * matrixB[k][j];
            }
        }
    }

    std::cout << "\nРезультат умножения (A * B):" << std::endl;
    printMatrix(resultMatrix);

    return 0;
}
```
