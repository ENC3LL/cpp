# Обход матрицы по спирали

Задача, проверяющая умение аккуратно работать с границами. Мы обходим матрицу по внешнему контуру, затем сдвигаем границы "внутрь" и повторяем, пока границы не сойдутся.

```c++
#include <iostream>
#include <vector>
#include <iomanip>

using namespace std;

int main() {
    int rows, cols;
    cout << "Введите размеры матрицы (строки столбцы): ";
    cin >> rows >> cols;

    vector<vector<int>> matrix(rows, vector<int>(cols));
    cout << "Заполните матрицу (для удобства можно по порядку):" << endl;
    for (int i = 0; i < rows; ++i) for (int j = 0; j < cols; ++j) cin >> matrix[i][j];

    cout << "\nОбход по спирали: ";
    
    int top = 0, bottom = rows - 1;
    int left = 0, right = cols - 1;

    while (top <= bottom && left <= right) {
        // 1. Движемся вправо по верхней границе
        for (int j = left; j <= right; ++j) {
            cout << matrix[top][j] << " ";
        }
        top++; // Сдвигаем верхнюю границу вниз

        // 2. Движемся вниз по правой границе
        for (int i = top; i <= bottom; ++i) {
            cout << matrix[i][right] << " ";
        }
        right--; // Сдвигаем правую границу влево

        // 3. Движемся влево по нижней границе (если она не совпала с верхней)
        if (top <= bottom) {
            for (int j = right; j >= left; --j) {
                cout << matrix[bottom][j] << " ";
            }
            bottom--; // Сдвигаем нижнюю границу вверх
        }

        // 4. Движемся вверх по левой границе (если она не совпала с правой)
        if (left <= right) {
            for (int i = bottom; i >= top; --i) {
                cout << matrix[i][left] << " ";
            }
            left++; // Сдвигаем левую границу вправо
        }
    }
    cout << endl;

    return 0;
}
```
