# Поворот матрицы на 90 градусов по часовой стрелке

Это популярная задача на собеседованиях. Эффективный способ сделать это "на месте" (без создания новой матрицы) для квадратной матрицы состоит из двух шагов:

1. **Транспонировать** матрицу.
2. **"Отразить"** (реверсировать) каждую строку.

```c++
#include <iostream>
#include <vector>
#include <algorithm> // для std::reverse
#include <iomanip>

void printMatrix(const std::vector<std::vector<int>>& matrix) {
    for (const auto& row : matrix) {
        for (int element : row) std::cout << std::setw(4) << element;
        std::cout << std::endl;
    }
}

int main() {
    int n;
    std::cout << "Введите размер квадратной матрицы (n x n): ";
    std::cin >> n;

    std::vector<std::vector<int>> matrix(n, std::vector<int>(n));
    std::cout << "Введите элементы матрицы:" << std::endl;
    for (int i = 0; i < n; ++i) for (int j = 0; j < n; ++j) std::cin >> matrix[i][j];

    std::cout << "\nИсходная матрица:" << std::endl;
    printMatrix(matrix);

    // Шаг 1: Транспонирование
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            std::swap(matrix[i][j], matrix[j][i]);
        }
    }

    // Шаг 2: Реверс каждой строки
    for (int i = 0; i < n; ++i) {
        std::reverse(matrix[i].begin(), matrix[i].end());
    }

    std::cout << "\nМатрица после поворота на 90 градусов:" << std::endl;
    printMatrix(matrix);

    return 0;
}
```
