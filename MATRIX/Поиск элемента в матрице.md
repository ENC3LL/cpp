# Поиск элемента в матрице

Это простой перебор всех элементов до тех пор, пока нужный не будет найден.

```c++
#include <iostream>
#include <vector>

// Функция поиска элемента. Возвращает пару {строка, столбец}
// или {-1, -1}, если элемент не найден.
std::pair<int, int> findElement(const std::vector<std::vector<int>>& matrix, int value) {
    for (int i = 0; i < matrix.size(); ++i) {
        for (int j = 0; j < matrix[i].size(); ++j) {
            if (matrix[i][j] == value) {
                return {i, j}; // Элемент найден
            }
        }
    }
    return {-1, -1}; // Элемент не найден
}

int main() {
    std::vector<std::vector<int>> matrix = {{10, 25, 13}, {41, 5, 96}, {7, 88, 19}};
    int value_to_find = 96;

    std::pair<int, int> position = findElement(matrix, value_to_find);

    if (position.first != -1) {
        std::cout << "Элемент " << value_to_find << " найден в позиции: ("
                  << position.first << ", " << position.second << ")" << std::endl;
    } else {
        std::cout << "Элемент " << value_to_find << " не найден." << std::endl;
    }

    return 0;
}
```
