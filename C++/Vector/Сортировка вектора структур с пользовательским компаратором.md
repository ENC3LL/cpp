# Сортировка вектора структур с пользовательским компаратором

Стандартная сортировка работает для простых типов (int, double). Но что если у вас вектор объектов? `sort` может принять третий аргумент — функцию-компаратор (чаще всего лямбда-функцию), которая объясняет, как сравнивать два объекта.

```c++
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

// Создадим структуру для представления пользователя
struct User {
    string name;
    int score;
};

int main() {
    vector<User> users;
    int num_users;

    cout << "Введите количество пользователей: ";
    cin >> num_users;

    for (int i = 0; i < num_users; ++i) {
        string name;
        int score;
        cout << "Введите имя и очки пользователя " << i + 1 << ": ";
        cin >> name >> score;
        users.push_back({name, score});
    }

    // --- Сортировка по очкам (по убыванию) с помощью лямбда-функции ---
    sort(users.begin(), users.end(), [](const User& a, const User& b) {
        // Возвращаем true, если 'a' должно идти ПЕРЕД 'b'
        return a.score > b.score; 
    });

    cout << "\nПользователи, отсортированные по очкам (убывание):" << endl;
    for (const auto& user : users) {
        cout << user.name << " - " << user.score << endl;
    }

    // --- Сортировка по имени (в алфавитном порядке) ---
    sort(users.begin(), users.end(), [](const User& a, const User& b) {
        return a.name < b.name;
    });

    cout << "\nПользователи, отсортированные по имени (алфавитный порядок):" << endl;
    for (const auto& user : users) {
        cout << user.name << " - " << user.score << endl;
    }

    return 0;
}
```
