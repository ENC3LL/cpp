# Создание и инициализация вектора

Векторы можно создавать пустыми, с заданным размером или сразу с начальными значениями.

```c++
#include <iostream>
#include <vector>

int main() {
    // 1. Создание пустого вектора
    std::vector<int> vec1;
    std::cout << "vec1 пуст. Размер: " << vec1.size() << std::endl;

    // 2. Создание вектора из n элементов, инициализированных значением
    int n_elements;
    int value;
    std::cout << "Введите количество элементов для vec2: ";
    std::cin >> n_elements;
    std::cout << "Введите значение для инициализации: ";
    std::cin >> value;
    std::vector<int> vec2(n_elements, value);

    std::cout << "vec2: ";
    for (int el : vec2) {
        std::cout << el << " ";
    }
    std::cout << std::endl;

    // 3. Создание вектора с помощью списка инициализации (C++11)
    std::vector<int> vec3 = {10, 20, 30, 40, 50};
    std::cout << "vec3 (список инициализации): ";
    for (int el : vec3) {
        std::cout << el << " ";
    }
    std::cout << std::endl;

    // 4. Создание копии другого вектора
    std::vector<int> vec4 = vec3;
    std::cout << "vec4 (копия vec3): ";
    for (int el : vec4) {
        std::cout << el << " ";
    }
    std::cout << std::endl;

    return 0;
}
```
