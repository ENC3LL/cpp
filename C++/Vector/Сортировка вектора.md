# Сортировка вектора

Вот способы сортировки вектора в C++:

## Базовая сортировка (по возрастанию)

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    vector<int> vec = {5, 2, 8, 1, 9};
    
    sort(vec.begin(), vec.end());
    
    for (int i = 0; i < vec.size(); i++) {
        cout << vec[i] << " "; // 1 2 5 8 9
    }
    return 0;
}
```

## Сортировка по убыванию

### Способ 1: Используя greater<>
```cpp
vector<int> vec = {5, 2, 8, 1, 9};
sort(vec.begin(), vec.end(), greater<int>());

// Результат: 9 8 5 2 1
```

### Способ 2: Используя лямбда-функцию
```cpp
vector<int> vec = {5, 2, 8, 1, 9};
sort(vec.begin(), vec.end(), [](int a, int b) {
    return a > b;
});

// Результат: 9 8 5 2 1
```

## Сортировка пользовательских структур

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

struct Person {
    string name;
    int age;
};

int main() {
    vector<Person> people = {{"Alice", 25}, {"Bob", 20}, {"Charlie", 30}};
    
    // Сортировка по возрасту (по возрастанию)
    sort(people.begin(), people.end(), [](const Person& a, const Person& b) {
        return a.age < b.age;
    });
    
    // Сортировка по имени (алфавитный порядок)
    sort(people.begin(), people.end(), [](const Person& a, const Person& b) {
        return a.name < b.name;
    });
    
    return 0;
}
```

## Частичная сортировка

```cpp
vector<int> vec = {9, 3, 6, 1, 7, 2, 8, 4, 5};

// Сортируем только первые 5 элементов
partial_sort(vec.begin(), vec.begin() + 5, vec.end());

// Результат: 1 2 3 4 5 9 8 7 6 (первые 5 отсортированы)
```

## Сортировка строк

```cpp
vector<string> words = {"banana", "apple", "cherry", "date"};

sort(words.begin(), words.end()); // Алфавитный порядок

// Результат: apple banana cherry date
```

## Обратная сортировка существующего вектора

```cpp
vector<int> vec = {1, 2, 3, 4, 5};
reverse(vec.begin(), vec.end());

// Результат: 5 4 3 2 1
```

## Проверка отсортированности

```cpp
vector<int> vec = {1, 2, 3, 4, 5};
bool isSorted = is_sorted(vec.begin(), vec.end());
cout << isSorted; // 1 (true)
```

## Полный пример с вводом и сортировкой

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> vec(n);
    
    for (int i = 0; i < n; i++) {
        cin >> vec[i];
    }
    
    // Сортировка по возрастанию
    sort(vec.begin(), vec.end());
    
    cout << "По возрастанию: ";
    for (int x : vec) {
        cout << x << " ";
    }
    cout << endl;
    
    // Сортировка по убыванию
    sort(vec.begin(), vec.end(), greater<int>());
    
    cout << "По убыванию: ";
    for (int x : vec) {
        cout << x << " ";
    }
    
    return 0;
}
```

**Важно:** Не забудьте `#include <algorithm>` для использования функции `sort()`!
