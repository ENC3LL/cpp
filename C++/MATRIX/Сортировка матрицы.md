# Сортировка матрицы

"Сортировка матрицы" — неоднозначное понятие. Можно отсортировать каждую строку отдельно или "выпрямить" матрицу в один вектор, отсортировать его и собрать обратно.

```c++
#include <iostream>
#include <vector>
#include <algorithm> // для std::sort

// ... (вставьте сюда функцию printMatrix) ...

int main() {
    std::vector<std::vector<int>> matrix = {{9, 2, 5}, {7, 1, 8}, {4, 6, 3}};

    // Способ 1: Сортировка каждой строки
    std::vector<std::vector<int>> sorted_rows_matrix = matrix;
    for (int i = 0; i < sorted_rows_matrix.size(); ++i) {
        std::sort(sorted_rows_matrix[i].begin(), sorted_rows_matrix[i].end());
    }
    std::cout << "Матрица с отсортированными строками:" << std::endl;
    printMatrix(sorted_rows_matrix);

    std::cout << "\n";

    // Способ 2: Полная сортировка (разборка, сортировка, сборка)
    std::vector<int> flat_vector;
    // Разборка
    for (int i = 0; i < matrix.size(); ++i) {
        for (int j = 0; j < matrix[i].size(); ++j) {
            flat_vector.push_back(matrix[i][j]);
        }
    }

    // Сортировка
    std::sort(flat_vector.begin(), flat_vector.end());

    // Сборка обратно в матрицу
    std::vector<std::vector<int>> fully_sorted_matrix = matrix;
    int k = 0;
    for (int i = 0; i < fully_sorted_matrix.size(); ++i) {
        for (int j = 0; j < fully_sorted_matrix[i].size(); ++j) {
            fully_sorted_matrix[i][j] = flat_vector[k++];
        }
    }
    std::cout << "Полностью отсортированная матрица:" << std::endl;
    printMatrix(fully_sorted_matrix);

    return 0;
}
```
