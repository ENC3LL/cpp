# Доступ к элементам и итерация

Доступ к элементам можно получить по индексу или через специальные методы. Итерация (перебор) — одна из самых частых операций.

```c++
#include <iostream>
#include <vector>

int main() {
    std::vector<int> vec = {5, 15, 25, 35, 45};

    // --- Доступ по индексу ---
    int index;
    std::cout << "Введите индекс для доступа к элементу (0-" << vec.size() - 1 << "): ";
    std::cin >> index;

    if (index >= 0 && index < vec.size()) {
        std::cout << "Элемент по индексу " << index << " (через []): " << vec[index] << std::endl;
        // Безопасный доступ с проверкой выхода за границы (генерирует исключение)
        std::cout << "Элемент по индексу " << index << " (через at()): " << vec.at(index) << std::endl;
    } else {
        std::cout << "Индекс вне диапазона!" << std::endl;
    }

    // --- Доступ к первому и последнему элементам ---
    if (!vec.empty()) {
        std::cout << "Первый элемент (front()): " << vec.front() << std::endl;
        std::cout << "Последний элемент (back()): " << vec.back() << std::endl;
    }

    // --- Перебор элементов ---
    // 1. Современный range-based for loop
    std::cout << "Перебор (range-based for): ";
    for (int element : vec) {
        std::cout << element << " ";
    }
    std::cout << std::endl;

    // 2. Классический цикл с итераторами
    std::cout << "Перебор (итераторы): ";
    for (auto it = vec.begin(); it != vec.end(); ++it) {
        std::cout << *it << " "; // *it - разыменование итератора для получения значения
    }
    std::cout << std::endl;
    
    return 0;
}
```
