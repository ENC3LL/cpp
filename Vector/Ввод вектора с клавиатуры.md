# Ввод вектора с клавиатуры

### Способ 1: С предварительным указанием размера
```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n;
    cin >> n; // Вводим количество элементов
    vector<int> vec(n);
    
    for (int i = 0; i < n; i++) {
        cin >> vec[i]; // Последовательно вводим элементы
    }
    
    // Проверка вывода
    for (int i = 0; i < n; i++) {
        cout << vec[i] << " ";
    }
    return 0;
}
```

### Способ 2: Без указания размера (до первого нечислового ввода)
```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    vector<int> vec;
    int temp;
    
    while (cin >> temp) {
        vec.push_back(temp); // Элементы добавляются пока ввод корректен
    }
    
    // Проверка вывода
    for (int i = 0; i < vec.size(); i++) {
        cout << vec[i] << " ";
    }
    return 0;
}
```

### Способ 3: Ввод одной строки чисел
```cpp
#include <iostream>
#include <vector>
#include <sstream>
using namespace std;

int main() {
    string line;
    getline(cin, line); // Читаем всю строку
    stringstream ss(line);
    
    vector<int> vec;
    int temp;
    while (ss >> temp) {
        vec.push_back(temp); // Разбираем строку на числа
    }
    
    // Проверка вывода
    for (int i = 0; i < vec.size(); i++) {
        cout << vec[i] << " ";
    }
    return 0;
}
```

### Примеры использования:
1. **Способ 1** (с размером):
   ```
   Ввод: 
   5
   10 20 30 40 50
   Вывод: 10 20 30 40 50
   ```

2. **Способ 2** (до нечислового ввода):
   ```
   Ввод: 
   10 20 30 x
   Вывод: 10 20 30
   ```

3. **Способ 3** (одна строка):
   ```
   Ввод:
   10 20 30 40 50
   Вывод: 10 20 30 40 50
   ```

Выберите подходящий способ в зависимости от формата входных данных.
