# Добавление, вставка и удаление элементов

Динамическая природа векторов позволяет легко управлять их содержимым: добавлять элементы в конец, вставлять в любую позицию и удалять.

```c++
#include <iostream>
#include <vector>

int main() {
    std::vector<int> vec = {10, 20, 30};

    // --- Добавление в конец ---
    int new_element_back;
    std::cout << "Введите элемент для добавления в конец: ";
    std::cin >> new_element_back;
    vec.push_back(new_element_back); // Добавляет элемент в конец

    std::cout << "После push_back: ";
    for (int el : vec) std::cout << el << " ";
    std::cout << std::endl;

    // --- Удаление с конца ---
    vec.pop_back(); // Удаляет последний элемент
    std::cout << "После pop_back: ";
    for (int el : vec) std::cout << el << " ";
    std::cout << std::endl;

    // --- Вставка элемента в середину ---
    int new_element_middle;
    int position;
    std::cout << "Введите элемент для вставки: ";
    std::cin >> new_element_middle;
    std::cout << "Введите позицию для вставки (индекс): ";
    std::cin >> position;

    if (position >= 0 && position <= vec.size()) {
        vec.insert(vec.begin() + position, new_element_middle); // Вставка элемента
    } else {
        std::cout << "Неверная позиция!" << std::endl;
    }
    
    std::cout << "После insert: ";
    for (int el : vec) std::cout << el << " ";
    std::cout << std::endl;
    
    // --- Удаление элемента по позиции ---
    std::cout << "Введите позицию для удаления (индекс): ";
    std::cin >> position;

     if (position >= 0 && position < vec.size()) {
        vec.erase(vec.begin() + position); // Удаление элемента
    } else {
        std::cout << "Неверная позиция!" << std::endl;
    }

    std::cout << "После erase: ";
    for (int el : vec) std::cout << el << " ";
    std::cout << std::endl;

    // --- Очистка вектора ---
    vec.clear(); // Удаляет все элементы
    std::cout << "После clear размер вектора: " << vec.size() << std::endl;
    
    return 0;
}
```
