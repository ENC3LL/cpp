# Поиск позиции вставки

Даны массив отсортированных целочисленных уникальных значений `nums` и целое число `target`, верните индекс `target`, если оно есть в `nums`. Иначе верните индекс позиции, куда его можно было бы вставить, сохранив порядок сортировки.

Вы должны написать алгоритм с временной сложностью `O(log n)`.

**Пример 1:**

**Ввод:** nums = [1,3,5,6], target = 5
**Вывод:** 2**Пример 2:**

**Ввод:** nums = [1,3,5,6], target = 2
**Вывод:** 1**Пример 3:**

**Ввод:** nums = [1,3,5,6], target = 7
**Вывод:** 4

# Решение

```c++
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {

        auto it = find(nums.begin(), nums.end(), target);

        if (it != nums.end()) {
            long long index = distance(nums.begin(), it);
            return index;
        }
        else {
            nums.push_back(target);
            sort(nums.begin(), nums.end());
            auto it = find(nums.begin(), nums.end(), target);
            long long index = distance(nums.begin(), it);
            return index;
        }

    }
};
```
