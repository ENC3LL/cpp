# Генерация скобочных последовательностей

Дано целое число `n`, которое представляет собой количество пар скобок. Напишите функцию, которая *генерирует все комбинации правильных скобочных последовательностей*.

**Пример 1:**

**Ввод:** n = 3
**Вывод:** ["((()))","(()())","(())()","()(())","()()()"]**Пример 2:**

**Ввод:** n = 1
**Вывод:** ["()"]


# Решение
```c++
class Solution {
public:
    void solve(int open, int close, string current, vector<string>& result, int n) {
        if (current.size() == 2 * n) {
            result.push_back(current);
            return;
        }

        if (open < n) {
            solve(open + 1, close, current + "(", result, n);
        }

        if (close < open) {
            solve(open, close + 1, current + ")", result, n);
        }
    }

    vector<string> generateParenthesis(int n) {
        vector<string> result;
        solve(0, 0, "", result, n);
        return result;
    }
};
```
